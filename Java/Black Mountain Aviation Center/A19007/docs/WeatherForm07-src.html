288                          public void updateTextBoxes()
289              {
290                    if (intAirport > -1 && intAirport < intAirportMax)
291                 {
292                    if (Integer.parseInt(strAltitude) > 0)
293                    {   
294                   
295    
296                     lblSp.setText(wea[intAirport].getWindSpeed(strAltitude));
297                       lblTemp.setText(wea[intAirport].getWindTemperature(strAltitude));
298                      lblDir.setText(wea[intAirport].getWindDir(strAltitude));
299                      lblAirport.setText("Airport="+strApt[intAirport]);
300                       lblAltitude.setText("Altitude="+strAltitude+"000");
301                   }
302                 }       
303              }

305               private class ButtonListener implements ActionListener
306              {
307                  public void actionPerformed(ActionEvent event)
308                 {
309                    /** 
310                        This button listener could be used for several
311                        buttons on the form.    */
312                    if (event.getSource() == btnExit)
313                    {
314                      
315                       System.exit(0);         
316                    }
317                 }
318              }  
319           
320              
321               private class AirportListener implements ActionListener
322              {
323                  public void actionPerformed(ActionEvent event)
324                 {
325                     intAirport = 0;
326                        
327                      for(int count=0; count<21; count++)
328                    {
329                      if(event.getSource() == Apt[count])
330                       {
331                         intAirport = count;
332                      }
333                    }
334                        
335                     updateTextBoxes();
336                 }
337             }
338                                
339               
340               private class AltitudeListener implements ActionListener
341              {
342                  public void actionPerformed(ActionEvent event)
343                 {
344                   
345                    strAltitude = event.getActionCommand();
346                    updateTextBoxes();
347                 }
348              }
